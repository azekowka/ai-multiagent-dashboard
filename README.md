#Multiagent Dashboard
1. cd backend
2. python -m venv venv
3. .\venv\Scripts\Activate.ps1
4. pip install -r requirements.txt
5. переименуйте .env.example в .env и вставьте ключи
6. python playground.py
7. cd ..
8. cd frontend
9. pnpm i
10. pnpm dev
